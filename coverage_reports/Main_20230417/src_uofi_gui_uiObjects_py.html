172 def BuildButtons(self, 

173 jsonObj: Dict = {}, 

174 jsonPath: str = "") -> None: 

175 """Builds a dictionary of Extron Buttons from a json object or file 

176 

177 Args (only one json arg required, jsonObj takes precedence over jsonPath): 

178 jsonObj (Dict, optional): The json object containing button information. 

179 Defaults to {}. 

180 jsonPath (str, optional): The path to the file containing json formatted 

181 button information. Defaults to "". 

182 

183 Raises: 

184 ValueError: if specified file at jsonPath does not exist 

185 ValueError: if neither jsonObj or jsonPath are specified 

186 """ 

187 

188 ## do not expect both jsonObj and jsonPath 

189 ## jsonObj should take priority over jsonPath 

190 if jsonObj == {} and jsonPath != "": # jsonObj is empty and jsonPath not blank 

191 if File.Exists(jsonPath): # jsonPath is valid, so load jsonObj from path 

192 jsonFile = File(jsonPath) 

193 jsonStr = jsonFile.read() 

194 jsonFile.close() 

195 jsonObj = json.loads(jsonStr) 

196 else: ## jsonPath was invalid, so return none (error) 

197 raise ValueError('Specified file does not exist') 

198 elif jsonObj == {} and jsonPath == "": 

199 raise ValueError('Either jsonObj or jsonPath must be specified') 

200 

201 ## format button info into self.Btns 

202 for button in jsonObj['buttons']: 

203 btnName = button['Name'] 

204 button.pop('Name') 

205 self.Btns[btnName] = Button(self, **button) 

206 self.Btns[btnName].holdTime = button['holdTime'] 

207 self.Btns[btnName].repeatTime = button['repeatTime'] 

208 

209 self.Btns[btnName].SetState(0) 

210 

211 @event(self.Btns[btnName], 'Pressed') 

212 def TempBtnHandler(button: 'Button', event: str): # pragma: no cover 

213 Log('Unconfigured Press Action: {} ({}, {})'.format(button.Name, button.ID, button))
